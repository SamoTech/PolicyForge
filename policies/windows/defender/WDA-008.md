## PowerShell Audit → Enforce Workflow

```powershell
# Step 1: Set all rules to Audit mode
$rules = @(
    "d4f940ab-401b-4efc-aadc-ad5f3c50688a",
    "3b576869-a4ec-4529-8536-b80a7769e899",
    "75668c1f-73b5-4cf0-bb93-3ecf5cb7cc84",
    "92e97fa1-2edf-4476-bdd6-9dd0b4dddc7b",
    "5beb7efe-fd9a-4556-801d-275e5ffc04cc",
    "d3e037e1-3eb8-44c8-a917-57927947596d",
    "be9ba2d9-53ea-4cdc-84e5-9b1eeee46550",
    "9e6c4e1f-7d60-472f-ba1a-a39ef669e4b0",
    "d1e49aac-8f56-4280-b9ba-993a6d77406c",
    "c1db55ab-c21a-4637-bb3f-a12568109d35",
    "e6db77e5-3df2-4cf1-b95a-636979351e5b"
)
$rules | ForEach-Object { Add-MpPreference -AttackSurfaceReductionRules_Ids $_ -AttackSurfaceReductionRules_Actions AuditMode }

# Step 2: Check audit events after 1 week
Get-WinEvent -LogName "Microsoft-Windows-Windows Defender/Operational" |
    Where-Object { $_.Id -in @(1121, 1122) } | Select TimeCreated, Id, Message

# Step 3: Switch to Block mode
$rules | ForEach-Object { Add-MpPreference -AttackSurfaceReductionRules_Ids $_ -AttackSurfaceReductionRules_Actions Enabled }
```
