## PowerShell — Defender Health Dashboard

```powershell
# Full Defender health report
$status = Get-MpComputerStatus
$prefs  = Get-MpPreference

$report = [PSCustomObject]@{
    ComputerName              = $env:COMPUTERNAME
    AMRunningMode             = $status.AMRunningMode
    RealTimeProtection        = $status.RealTimeProtectionEnabled
    BehaviorMonitor           = $status.BehaviorMonitorEnabled
    CloudProtection           = $status.AntispywareEnabled
    TamperProtected           = $status.IsTamperProtected
    SignatureAge              = $status.AntivirusSignatureAge
    SignatureVersion          = $status.AntivirusSignatureVersion
    LastQuickScan             = $status.QuickScanStartTime
    LastFullScan              = $status.FullScanStartTime
    PUAProtection             = $prefs.PUAProtection
    NetworkProtection         = $prefs.EnableNetworkProtection
    ControlledFolderAccess    = $prefs.EnableControlledFolderAccess
    ExclusionPaths            = ($prefs.ExclusionPath -join "; ")
}

$report | Format-List
$report | Export-Csv "C:\PolicyForge\defender-health-$(Get-Date -Format yyyyMMdd).csv" -NoTypeInformation
```
